<programa> ::= <instrução> ";" <programa> | <instrução> ";"
<instrução> ::= <atribuição> | <declaração> | <condicional> | <laço> | <impressão>
<atribuição> ::= <identificador> "=" <expressão>
<declaração> ::= "int" <identificador> | "float" <identificador>
<condicional> ::= "if" "(" <condição> ")" "{" <programa> "}"
<laço> ::= "while" "(" <condição> ")" "{" <programa> "}"
<impressão> ::= "print" "(" <identificador> ")"
<expressão> ::= <termo> | <termo> <operador> <expressão>
<condição> ::= <expressão> <operador-relacional> <expressão>
<termo> ::= <identificador> | <número>
<identificador> ::= [a-zA-Z_][a-zA-Z0-9_]*
<número> ::= [0-9]+ | [0-9]+"."[0-9]+
<operador> ::= "+" | "-" | "*" | "/"
<operador-relacional> ::= ">" | "<" | "==" | "!=" | ">=" | "<="
